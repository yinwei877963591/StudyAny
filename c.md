数据类型：

```c++
using namespace std;

int main()
{
  cout << "Size of char :" << sizeof(char) << endl;
  cout << "Size of char :" << sizeof(int) << endl;
  cout << "Size of char :" << sizeof(short int) << endl;
  cout << "Size of char :" << sizeof(long int) << endl;
  cout << "Size of char :" << sizeof(float) << endl;
  cout << "Size of char :" << sizeof(double) << endl;
  return 0;
}
```
